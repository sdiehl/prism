-- `with f <- handler { .. }` binds a first-class handler instance. It desugars
-- to a fresh private effect whose operations are renamed (unforgeable from
-- source); `f.op(args)` targets that instance regardless of any nearer handler.
with conf <- handler
  read(k) => k("conf.toml")
  return r => r
println(conf.read())
  ==>
handle
  println(do read@conf())            -- conf.read() performs the private op
with
  read@conf(k) => k("conf.toml")
  return r => r
