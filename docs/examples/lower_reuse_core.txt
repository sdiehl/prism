fn bump(xs) =
  return xs to t@735
  case t@735 of
    Nil =>
      reuse_token t@735 to reuse#t@735
      reuse reuse#t@735 as Nil
    Cons(h, t) =>
      dup h
      dup t
      reuse_token t@735 to reuse#t@735
      return h to t@736
      return 1 to t@737
      t@736 + t@737 to t@738
      return t to t@739
      bump(t@739) to t@740
      reuse reuse#t@735 as Cons(t@738, t@740)
